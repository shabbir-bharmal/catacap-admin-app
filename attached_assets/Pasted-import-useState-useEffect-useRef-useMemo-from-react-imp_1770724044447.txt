// Helper function to proxy external URLs through backend to avoid CORS issues
function getProxyUrl(url: string | undefined): string | undefined {
  if (!url) return undefined;
  // Only proxy external Azure blob storage URLs
  if (url.includes('catacapstorage.blob.core.windows.net')) {
    return `/api/proxy-file?url=${encodeURIComponent(url)}`;
  }
  return url;
}

function Speedometer({ percentage }: { percentage: number }) {
  const clampedPct = Math.min(Math.max(percentage, 0), 100);

  const cx = 50;
  const cy = 50;
  const radius = 40;

  const startX = cx - radius;
  const startY = cy;
  const endX = cx + radius;
  const endY = cy;

  const arcPath = `M ${startX} ${startY} A ${radius} ${radius} 0 1 1 ${endX} ${endY}`;

  return (
    <div className="relative w-28 h-28 flex items-center justify-center mx-auto">
      <svg viewBox="0 0 100 100" className="w-full h-full">
        <path
          d={arcPath}
          fill="none"
          stroke="#e2e8f0"
          strokeWidth="8"
          strokeLinecap="round"
        />
        <motion.path
          d={arcPath}
          fill="none"
          stroke="#80b73f"
          strokeWidth="8"
          strokeLinecap="round"
          initial={{ pathLength: 0 }}
          whileInView={{ pathLength: clampedPct / 100 }}
          viewport={{ once: true }}
          transition={{ duration: 1, delay: 0.1 }}
          style={{
            strokeDasharray: "1",
            strokeDashoffset: "0",
          }}
        />
        <text
          x={cx}
          y={cy + 2}
          textAnchor="middle"
          dominantBaseline="middle"
          fontSize="18"
          fontWeight="700"
          fill="#80b73f"
        >
          {percentage}%
        </text>
        <text
          x={cx}
          y={cy + 16}
          textAnchor="middle"
          dominantBaseline="middle"
          fontSize="9"
          fill="#6b7280"
        >
          funded
        </text>
      </svg>
    </div>
  );
}

function PitchMaterialSection({ pdfUrl }: { pdfUrl?: string }) {
  const [numPages, setNumPages] = useState<number>(0);
  const [pageNumber, setPageNumber] = useState(1);
  const [scale, setScale] = useState(1.0);
  const [pdfError, setPdfError] = useState(false);
  const [isLoading, setIsLoading] = useState(true);

  const displayPdfUrl = getProxyUrl(pdfUrl) || "/sample-pitch.pdf";

  const pdfOptions = useMemo(
    () => ({
      cMapUrl: "https://unpkg.com/pdfjs-dist@4.4.168/cmaps/",
      cMapPacked: true,
    }),
    [],
  );

  useEffect(() => {
    const handleUnhandledRejection = (event: PromiseRejectionEvent) => {
      if (
        event.reason?.name === "AbortError" ||
        event.reason?.message?.includes("aborted") ||
        event.reason?.message?.includes("BodyStreamBuffer")
      ) {
        event.preventDefault();
      }
    };

    window.addEventListener("unhandledrejection", handleUnhandledRejection);
    return () =>
      window.removeEventListener(
        "unhandledrejection",
        handleUnhandledRejection,
      );
  }, []);

  function onDocumentLoadSuccess({ numPages }: { numPages: number }) {
    setNumPages(numPages);
    setPdfError(false);
    setIsLoading(false);
  }

  function onDocumentLoadError(error: Error) {
    if (error.name === "AbortError" || error.message?.includes("aborted")) {
      return;
    }
    setPdfError(true);
    setIsLoading(false);
  }

  const goToPrevPage = () => setPageNumber((prev) => Math.max(prev - 1, 1));
  const goToNextPage = () =>
    setPageNumber((prev) => Math.min(prev + 1, numPages));
  const zoomIn = () => setScale((prev) => Math.min(prev + 0.2, 2.0));
  const zoomOut = () => setScale((prev) => Math.max(prev - 0.2, 0.5));

  return (
    <section data-testid="section-pitch-material">
      <div className="flex items-center justify-between mb-4">
        <h3 className="text-lg font-semibold text-foreground flex items-center gap-2">
          <FileText className="w-5 h-5" />
          Pitch Material
        </h3>
        <Button
          variant="outline"
          size="sm"
          asChild
          data-testid="button-download-pdf"
        >
          <a
            href={displayPdfUrl}
            download
            target="_blank"
            rel="noopener noreferrer"
          >
            <Download className="w-4 h-4 mr-2" />
            Download PDF
          </a>
        </Button>
      </div>

      <Card className="p-4" data-testid="card-pdf-viewer">
        {pdfError ? (
          <div className="flex flex-col items-center justify-center py-12 text-muted-foreground">
            <FileText className="w-12 h-12 mb-4 opacity-50" />
            <p className="text-sm">Unable to load PDF document</p>
            <p className="text-xs mt-1">
              Please try downloading the file instead
            </p>
          </div>
        ) : (
          <>
            <div className="flex items-center justify-between mb-4 pb-3 border-b">
              <div className="flex items-center gap-2">
                <Button
                  variant="outline"
                  size="icon"
                  onClick={goToPrevPage}
                  disabled={pageNumber <= 1}
                  data-testid="button-prev-page"
                >
                  <ChevronLeft className="w-4 h-4" />
                </Button>
                <span
                  className="text-sm text-muted-foreground min-w-[80px] text-center"
                  data-testid="text-page-info"
                >
                  Page {pageNumber} of {numPages || "..."}
                </span>
                <Button
                  variant="outline"
                  size="icon"
                  onClick={goToNextPage}
                  disabled={pageNumber >= numPages}
                  data-testid="button-next-page"
                >
                  <ChevronRight className="w-4 h-4" />
                </Button>
              </div>

              <div className="flex items-center gap-2">
                <Button
                  variant="outline"
                  size="icon"
                  onClick={zoomOut}
                  disabled={scale <= 0.5}
                  data-testid="button-zoom-out"
                >
                  <ZoomOut className="w-4 h-4" />
                </Button>
                <span className="text-sm text-muted-foreground min-w-[50px] text-center">
                  {Math.round(scale * 100)}%
                </span>
                <Button
                  variant="outline"
                  size="icon"
                  onClick={zoomIn}
                  disabled={scale >= 2.0}
                  data-testid="button-zoom-in"
                >
                  <ZoomIn className="w-4 h-4" />
                </Button>
              </div>
            </div>

            <div
              className="overflow-auto bg-muted/30 rounded-md flex justify-center min-h-[400px] max-h-[600px]"
              data-testid="pdf-container"
            >
              <Document
                file={displayPdfUrl}
                onLoadSuccess={onDocumentLoadSuccess}
                onLoadError={onDocumentLoadError}
                loading={
                  <div className="flex items-center justify-center py-20">
                    <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-primary"></div>
                  </div>
                }
                options={pdfOptions}
              >
                <Page
                  pageNumber={pageNumber}
                  scale={scale}
                  renderTextLayer={false}
                  renderAnnotationLayer={false}
                />
              </Document>
            </div>
          </>
        )}
      </Card>
    </section>
  );
}

function parseIds(ids: string): number[] {
  if (!ids) return [];
  return ids
    .split(",")
    .map((id) => parseInt(id.trim(), 10))
    .filter((id) => !isNaN(id));
}

function stripHtml(html: string): string {
  const doc = new DOMParser().parseFromString(html, "text/html");
  return doc.body.textContent || "";
}

                    className="text-sm font-medium text-foreground mb-3"
                    data-testid="label-donate-to-invest"
                  >
                    Donate to invest?
                  </p>

                  <div
                    className="grid grid-cols-3 gap-2 mb-4"
                    data-testid="preset-amounts"
                  >
                    {[1000, 2000, 5000, 10000, 25000].map((amount) => (
                      <Button
                        key={amount}
                        variant={
                          investAmount === String(amount)
                            ? "default"
                            : "outline"
                        }
                        size="sm"
                        onClick={() => setInvestAmount(String(amount))}
                        className="text-sm"
                        data-testid={`button-amount-${amount}`}
                      >
                        ${amount.toLocaleString()}
                      </Button>
                    ))}
                    <Button
                      variant={
                        !["1000", "2000", "5000", "10000", "25000"].includes(
                          investAmount,
                        )
                          ? "default"
                          : "outline"
                      }
                      size="sm"
                      onClick={() => setInvestAmount("")}
                      className="text-sm"
                      data-testid="button-amount-other"
                    >
                      OTHER
                    </Button>
                  </div>

                  <p className="text-sm text-muted-foreground mb-2">
                    Invest: Min (${investmentData.minimumInvestment})
                  </p>
                  <div className="flex gap-2 mb-3">
                    <div className="relative flex-1">
                      <span className="absolute left-3 top-1/2 -translate-y-1/2 text-foreground">
                        $
                      </span>
                      <Input
                        type="text"
                        value={investAmount}
                        onChange={(e) =>
                          setInvestAmount(e.target.value.replace(/[^0-9]/g, ""))
                        }
                        className="pl-7"
                        placeholder={investmentData.minimumInvestment}
                        data-testid="input-invest-amount"
                      />
                    </div>
                  </div>
                  <Button
                    className="w-full font-semibold"
                    data-testid="button-invest-now"
                  >
                    Invest Now
                  </Button>
                </div>

                <div className="flex gap-2 mt-4">
                  <Button
                    variant="outline"
                    className="flex-1 gap-2"
                    data-testid="button-share"
                  >
                    <Share2 className="w-4 h-4" />
                    Share
                  </Button>
                  <Button
                    variant="outline"
                    className="flex-1 gap-2"
                    data-testid="button-get-updates"
                  >
                    <Bell className="w-4 h-4" />
                    Get Updates
                  </Button>
                </div>
              </Card>

              {matchedCampaigns.length > 0 && (
                <Card className="p-5" data-testid="card-similar-investments">
                  <div className="flex items-center justify-between mb-4">
                    <h4
                      className="text-sm font-medium text-muted-foreground"
                      data-testid="heading-similar-investments"
                    >
                      Similar Investments
                    </h4>
                    <div className="flex items-center gap-1">
                      <Button
                        variant={
                          similarViewMode === "carousel" ? "default" : "ghost"
                        }
                        size="icon"
                        className="h-7 w-7"
                        onClick={() => setSimilarViewMode("carousel")}
                        data-testid="button-view-carousel"
                      >
                        <LayoutGrid className="w-4 h-4" />
                      </Button>
                      <Button
                        variant={
                          similarViewMode === "list" ? "default" : "ghost"
                        }
                        size="icon"
                        className="h-7 w-7"
                        onClick={() => setSimilarViewMode("list")}
                        data-testid="button-view-list"
                      >
                        <List className="w-4 h-4" />
                      </Button>
                    </div>
                  </div>

                  {similarViewMode === "carousel" && (
                    <div data-testid="carousel-similar-investments">
                      <div className="relative">
                        <AnimatePresence mode="wait">
                          <motion.div
                            key={carouselIndex}
                            initial={{ opacity: 0, x: 20 }}
                            animate={{ opacity: 1, x: 0 }}
                            exit={{ opacity: 0, x: -20 }}
                            transition={{ duration: 0.2 }}
                            className="relative pt-8 mt-8"
                            data-testid={`carousel-card-${matchedCampaigns[carouselIndex]?.id}`}
                          >
                            <div
                              className="absolute -top-4 left-4 w-16 h-16 rounded-full flex items-center justify-center text-white text-xl font-bold z-10 border-4 border-background overflow-hidden"
                              style={{ backgroundColor: "#7dd3fc" }}
                            >
                              {matchedCampaigns[carouselIndex]
                                ?.tileImageFileName ? (
                                <img
                                  src={getProxyUrl(
                                    matchedCampaigns[carouselIndex]
                                      .tileImageFileName
                                  )}
                                  alt={matchedCampaigns[carouselIndex].name}
                                  className="w-full h-full object-cover"
                                  onError={(e) => {
                                    (e.target as HTMLImageElement).style.display = 'none';
                                    (e.target as HTMLImageElement).parentElement!.textContent = matchedCampaigns[carouselIndex]?.name.charAt(0) || '';
                                  }}
                                />
                              ) : (
                                matchedCampaigns[carouselIndex]?.name.charAt(0)
                              )}
                            </div>

                            <div className="border border-border rounded-xl p-4 pt-6 bg-muted/20">
                              <MiniSpeedometer
                                percentage={Math.min(
                                  Math.round(
                                    (matchedCampaigns[carouselIndex]
                                      ?.currentBalance /
                                      (parseFloat(
                                        matchedCampaigns[carouselIndex]
                                          ?.target || "1",
                                      ) *
                                        1000)) *
                                      100,
                                  ),
                                  100,
                                )}
                              />

                              <div className="text-center mt-3">
                                <div className="flex items-center justify-center gap-1">
                                  <span className="text-lg font-bold text-[#80b73f]">
                                    $
                                    {matchedCampaigns[
                                      carouselIndex
                                    ]?.currentBalance.toLocaleString()}
                                  </span>
                                  <span className="text-xs text-muted-foreground">
                                    of $
                                    {(
                                      parseFloat(
                                        matchedCampaigns[carouselIndex]
                                          ?.target || "0",
                                      ) * 1000
                                    ).toLocaleString()}{" "}
                                    goal
                                  </span>
                                </div>

                                <p
                                  className="text-sm font-medium text-foreground mb-3"
                                  data-testid={`carousel-name-${matchedCampaigns[carouselIndex]?.id}`}
                                >
                                  {matchedCampaigns[carouselIndex]?.name}
                                </p>

                                <Button
                                  variant="outline"
                                  size="sm"
                                  className="w-full"
                                  data-testid={`button-view-details-${matchedCampaigns[carouselIndex]?.id}`}
                                >
                                  View Details
                                </Button>
                              </div>
                            </div>
                          </motion.div>
                        </AnimatePresence>

                        <div className="flex items-center justify-between mt-3">
                          <Button
                            variant="ghost"
                            size="icon"
                            className="h-8 w-8"
                            onClick={() =>
                              setCarouselIndex((prev) =>
                                prev === 0
                                  ? matchedCampaigns.length - 1
                                  : prev - 1,
                              )
                            }
                            data-testid="button-carousel-prev"
                          >
                            <ChevronLeft className="w-4 h-4" />
                          </Button>

                          <div className="flex items-center gap-1.5">
                            {matchedCampaigns.slice(0, 5).map((_, idx) => (
                              <button
                                key={idx}
                                className={`w-2 h-2 rounded-full transition-colors ${idx === carouselIndex ? "bg-primary" : "bg-muted-foreground/30"}`}
                                onClick={() => setCarouselIndex(idx)}
                                data-testid={`carousel-dot-${idx}`}
                              />
                            ))}
                          </div>

                          <Button
                            variant="ghost"
                            size="icon"
                            className="h-8 w-8"
                            onClick={() =>
                              setCarouselIndex((prev) =>
                                prev === matchedCampaigns.length - 1
                                  ? 0
                                  : prev + 1,
                              )
                            }
                            data-testid="button-carousel-next"
                          >
                            <ChevronRight className="w-4 h-4" />
                          </Button>
                        </div>
                      </div>
                    </div>
                  )}

                  {similarViewMode === "list" && (
                    <div
                      className="space-y-3"
                      data-testid="list-similar-investments"
                    >
                      {matchedCampaigns.slice(0, 5).map((campaign) => (
                        <div
                          key={campaign.id}
                          className="flex items-center gap-3 p-2 rounded-lg hover-elevate cursor-pointer"
                          data-testid={`similar-investment-${campaign.id}`}
                        >
                          <div
                            className="w-10 h-10 rounded-full flex items-center justify-center text-white text-sm font-bold shrink-0 overflow-hidden"
                            style={{ backgroundColor: "#80b73f" }}
                          >
                            {campaign.tileImageFileName ? (
                              <img
                                src={getProxyUrl(campaign.tileImageFileName)}
                                alt={campaign.name}
                                className="w-full h-full object-cover"
                                onError={(e) => {
                                  (e.target as HTMLImageElement).style.display = 'none';
                                  (e.target as HTMLImageElement).parentElement!.textContent = campaign.name.charAt(0);
                                }}
                              />
                            ) : (
                              campaign.name.charAt(0)
                            )}
                          </div>
                          <div className="flex-1 min-w-0">
                            <p
                              className="text-sm font-medium text-foreground truncate"
                              data-testid={`similar-name-${campaign.id}`}
                            >
                              {campaign.name}
                            </p>
                            <div className="flex items-center gap-2 text-xs text-muted-foreground">
                              <span>
                                {campaign.numberOfInvestors} investors
                              </span>
                            </div>
                          </div>
                          <div className="text-right shrink-0">
                            <div className="text-xs font-semibold text-[#80b73f]">
                              ${campaign.currentBalance.toLocaleString()}
                            </div>
                            <div className="text-[10px] text-muted-foreground">
                              raised
                            </div>
                          </div>
                        </div>
                      ))}
                    </div>
                  )}
                </Card>
              )}
            </div>
          </div>
        </div>
      </main>
    </div>
  );
}
